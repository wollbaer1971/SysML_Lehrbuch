\newacronym{mbse}{MBSE}{Model-Based Systems Engineering}
\newacronym{sysml}{SysML}{Systems Modeling Language}
\newacronym{ros2}{ROS2}{Robot Operating System 2}
\newacronym{bdd}{BDD}{Block Definition Diagram (SysML v1)}
\newacronym{ibd}{IBD}{Internal Block Diagram (SysML v1)}
\newacronym{slam}{SLAM}{Simultaneous Localization and Mapping}
\newacronym{ml}{ML}{Machine Learning}
\newacronym{api}{API}{Application Programming Interface}
\newacronym{qos}{QoS}{Quality of Service}

\newglossaryentry{requirement}{
    name=Anforderung,
    description={Beschreibung einer Fähigkeit, Eigenschaft oder Einschränkung, die ein System erfüllen muss}
}
\newglossaryentry{block}{
    name=Block,
    description={Strukturelement in SysML v1. In SysML v2 ersetzt durch \texttt{part def} (Typdefinition) und \texttt{part} (Instanz)}
}
\newglossaryentry{partdef}{
    name=part def,
    description={Typdefinition in SysML v2. Entspricht dem Block aus SysML v1. Definiert, welche Eigenschaften und Teile ein Systemelement besitzen kann}
}
\newglossaryentry{node}{
    name=ROS2-Node,
    description={Eigenständiger Prozess in ROS2, der Daten verarbeitet und über Topics, Services oder Actions kommuniziert}
}
\newglossaryentry{topic}{
    name=Topic,
    description={Nachrichtenkanal in ROS2 für publish-subscribe-Kommunikation}
}
\newglossaryentry{traceability}{
    name=Traceability,
    description={Nachvollziehbare Verbindung zwischen Anforderungen, Architektur, Implementierung und Tests im Systemmodell}
}
\newglossaryentry{middleware}{
    name=Middleware,
    description={Software-Schicht, die Kommunikation zwischen verteilten Prozessen (z.\,B. ROS2-Nodes) vermittelt, ohne dass diese die technischen Transportdetails der Gegenseite kennen müssen}
}
\newglossaryentry{cyberphysical}{
    name=cyber-physisches System,
    plural=cyber-physische Systeme,
    description={System, in dem physische Komponenten durch Software gesteuert werden und über Sensoren, Aktoren oder Netzwerke eng mit dieser Software verbunden sind}
}
\newglossaryentry{conjport}{
    name=konjugierter Port,
    description={Port, dessen Flussrichtung gegenüber der ursprünglichen Portdefinition umgekehrt ist (Kennzeichen: Tilde \textasciitilde); ein sendender Port wird dadurch zu einem empfangenden Port}
}
