\RedeclareSectionCommand[tocnumwidth=3.8em]{section}

% Inhaltsverzeichnis nur bis Subsection-Ebene. Subsubsection-Eintraege
% (3-stellige Nummern wie 2.12.3 + tief eingerueckter Titel) sprengen auf dem
% schmaleren 6x9in-Trim die Zeilenbreite des TOC und laufen ueber den rechten
% Rand. Nummerierung im Fliesstext bleibt unveraendert (nur \tocdepth, nicht
% \secnumdepth), Subsubsections werden weiterhin nummeriert, erscheinen nur
% nicht mehr im Verzeichnis.
\setcounter{tocdepth}{2}

% Stil fuer reine Text-/Baumdarstellungen (Verzeichnisbaeume, Ablaufskizzen),
% die wie \verbatim aussehen sollen, aber - anders als \verbatim - lange
% Zeilen umbrechen muessen, damit sie auf schmalen Trims (z. B. 6x9in)
% nicht ueber den Satzspiegel hinauslaufen.
\lstdefinestyle{plaintext}{
    style=code,
    numbers=none,
    frame=single,
}
