% Zusaetzlicher Justier-Spielraum: faengt Faelle auf, die microtype allein
% nicht mehr ausgleichen kann (lange deutsche Komposita, schmale Spalten bei
% kleineren Trims). Ohne diese Werte erzeugt LaTeX in solchen Faellen einen
% sichtbar ueber den Satzspiegel hinausragenden Overfull-hbox, statt die
% Zeile mit minimal mehr Wortabstand sauber zu brechen.

% Druckausgaben (KDP, BoD) setzen \druckausgabetrue VOR \input{preamble_common}.
% KDPs Pruefung markiert Seiten mit klickbaren Hyperlink-Annotationen
% (Querverweise, Glossarbegriffe, das komplette verlinkte Inhaltsverzeichnis)
% offenbar unabhaengig von deren Farbe als auffaellig - schwarze Linkfarbe
% allein hat die Meldungen nicht beseitigt. Fuer Druckausgaben laedt
% hyperref deshalb im draft-Modus: alle \href/\ref/\gls-Befehle bleiben
% nutzbar und setzen ihren Text normal, erzeugen aber keine PDF-Link-
% Annotation mehr. Die Arbeitsausgabe (main.tex) behaelt volle, klickbare
% Verlinkung fuer die Bildschirmnutzung.
\makeatletter
\@ifundefined{ifdruckausgabe}{\newif\ifdruckausgabe\druckausgabefalse}{}
\makeatother
\ifdruckausgabe
    \usepackage[draft]{hyperref}
\else
    \usepackage{hyperref}
\fi

\RedeclareSectionCommand[tocnumwidth=3.8em]{section}
\RedeclareSectionCommand[tocnumwidth=4.8em]{subsection}
\RedeclareSectionCommand[tocnumwidth=5.8em]{subsubsection}

% Inhaltsverzeichnis nur bis Section-Ebene. Bereits Subsection-Eintraege
% (3-stellige Nummern wie 1.10.1 + tief eingerueckter Titel, nicht erst
% Subsubsection - Zaehltiefe: chapter=0, section=1, subsection=2) sprengen
% auf dem schmaleren 6x9in-Trim die Zeilenbreite des TOC und laufen ueber
% den rechten Rand. tocdepth=2 (vorherige Einstellung) liess Subsections
% noch zu, das reichte nicht - deshalb jetzt tocdepth=1. Nummerierung im
% Fliesstext bleibt unveraendert (nur \tocdepth, nicht \secnumdepth),
% Subsections werden weiterhin nummeriert, erscheinen nur nicht mehr im
% Verzeichnis.
\setcounter{tocdepth}{1}

    breaklines=true,

    numbers=left,
    numberstyle=\tiny,
    xleftmargin=0.5cm,

    showstringspaces=false,
    tabsize=2,
    extendedchars=true,
    inputencoding=utf8,
    upquote=true,
    literate=
        {ä}{{\"a}}1 {ö}{{\"o}}1 {ü}{{\"u}}1
        {Ä}{{\"A}}1 {Ö}{{\"O}}1 {Ü}{{\"U}}1
        {ß}{{\ss{}}}1
}
\lstset{style=code}

% Stil fuer reine Text-/Baumdarstellungen (Verzeichnisbaeume, Ablaufskizzen),
% die wie \verbatim aussehen sollen, aber - anders als \verbatim - lange
% Zeilen umbrechen muessen, damit sie auf schmalen Trims (z. B. 6x9in)
% nicht ueber den Satzspiegel hinauslaufen.
\lstdefinestyle{plaintext}{
    style=code,
    numbers=none,
    frame=single,
}
